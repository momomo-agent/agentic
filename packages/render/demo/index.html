📦 Usage

<script src="agentic-render.js"></script>
<script>
// Create renderer
const r = AgenticRender.create('#output', { theme: 'dark' })

// Stream tokens one by one — no flicker
r.append('# Hello\n\n')
r.append('Some **bold** text\n\n')
r.append('```js\n')
r.append('console.log("hi")')  // partial code block
r.append('\n```')              // closes cleanly

// Or render all at once
r.set('# Complete markdown')

// Switch theme
r.setTheme('light')
</script>
