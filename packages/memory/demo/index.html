🧠 agentic-memory

AI memory — conversation context + knowledge retrieval

💬 Short-term: Conversation Context

📚 Long-term: Knowledge Retrieval

📦 Usage

<script src="agentic-memory.js"></script>
<script>
const { createMemory } = AgenticMemory

// Short-term: conversation context
const mem = createMemory({ maxTokens: 8000 })
await mem.user('What is quantum computing?')
await mem.assistant('Quantum computing uses qubits...')
const messages = mem.messages() // for LLM API

// Long-term: knowledge retrieval
const mem2 = createMemory({ knowledge: true })
await mem2.learn('physics', 'Quantum computing...')
const results = await mem2.recall('How do qubits work?')
</script>
