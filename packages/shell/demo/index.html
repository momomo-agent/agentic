💻 agentic-shell

Unix-like shell for AI — runs in browser with virtual filesystem

📦 Usage

<script src="agentic-shell.js"></script>
<script type="module">
import { AgenticFileSystem } from 'agentic-filesystem'
import { AgenticShell } from 'agentic-shell'

// Create virtual filesystem
const fs = new AgenticFileSystem()

// Create shell
const shell = new AgenticShell(fs)

// Execute commands
const result = await shell.exec('ls -la')
console.log(result.output)

// Pipe commands
await shell.exec('echo "Hello World" | grep Hello')

// File operations
await shell.exec('mkdir test')
await shell.exec('echo "content" > test/file.txt')
await shell.exec('cat test/file.txt')
</script>
