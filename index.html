A complete AI chat app with memory and streaming rendering.

Complete AI chat app
import { ask } from 'agentic-lite' import { createMemory } from 'agentic-memory' const mem = createMemory({ maxTokens: 8000 }) const view = AgenticRender.create('#output') async function chat(prompt) { await mem.user(prompt) const result = await ask(prompt, { provider: 'anthropic', apiKey: 'sk-ant-...', history: mem.history(), tools: ['search'], }) await mem.assistant(result.answer) view.set(result.answer) // render with streaming markdown } chat('What happened in AI this week?') chat('Tell me more about the first one') // has context!
